from elasticsearch import Elasticsearch

class ElasticsearchClient:
    _instance = None
    
    @classmethod
    def get_client(cls):
        if cls._instance is None:
            cls._instance = Elasticsearch(
                hosts=['http://localhost:9200'],
                timeout=30,
                max_retries=3,
                retry_on_timeout=True
            )
        return cls._instance
